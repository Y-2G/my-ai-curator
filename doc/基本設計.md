- **Framework**: Next.js 14 (App Router)
- **Language**: TypeScript
- **Styling**: Tailwind CSS
- **UI Components**: Headless UI / Radix UI
- **State Management**: React Query + Zustand
- **Build**: Static Site Generation (SSG) + Incremental Static Regeneration (ISR)

#### バックエンド
- **Runtime**: Node.js 18+
- **API**: Next.js API Routes
- **Language**: TypeScript
- **Validation**: Zod
- **Date Handling**: date-fns

#### データベース・ストレージ
- **Primary DB**: Vercel Postgres
- **Cache**: Vercel KV (Redis)
- **File Storage**: Vercel Blob
- **ORM**: Prisma

#### インフラ・デプロイ
- **Hosting**: Vercel
- **Domain**: Custom domain

### 2.2 主要テーブル定義

#### users テーブル
```sql
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    name VARCHAR(100) NOT NULL,
    profile JSONB DEFAULT '{}',
    interests JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

#### articles テーブル
```sql
CREATE TABLE articles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title TEXT NOT NULL,
    summary TEXT NOT NULL,
    content TEXT NOT NULL,
    category_id UUID REFERENCES categories(id),
    author_id UUID REFERENCES users(id),
    interest_score INTEGER DEFAULT 0,
    quality_score INTEGER DEFAULT 0,
    published_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    -- インデックス
    INDEX idx_articles_published_at (published_at DESC),
    INDEX idx_articles_category (category_id),
    INDEX idx_articles_interest_score (interest_score DESC)
);
```

#### sources テーブル
```sql
CREATE TABLE sources (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    article_id UUID REFERENCES articles(id) ON DELETE CASCADE,
    url TEXT NOT NULL,
    title VARCHAR(500),
    type VARCHAR(50) NOT NULL, -- 'news', 'reddit', 'github', 'rss'
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

```typescript
export async function authMiddleware(req: NextRequest) {
  const apiKey = req.headers.get('x-api-key');
  const path = req.nextUrl.pathname;
  
  // 内部API保護
  if (path.startsWith('/api/internal/')) {
    if (apiKey !== process.env.INTERNAL_API_KEY) {
      return new Response('Unauthorized', { status: 401 });
    }
  }
  
  // CORS対応
  const origin = req.headers.get('origin');
  if (origin && !allowedOrigins.includes(origin)) {
    return new Response('Forbidden', { status: 403 });
  }
  
  return NextResponse.next();
}
```

### 6.3 レート制限
```typescript
// Vercel KV を使用したレート制限
class RateLimiter {
  private redis: VercelKV;
  
  async checkLimit(
    key: string,
    limit: number,
    windowMs: number
  ): Promise<boolean> {
    const current = await this.redis.incr(key);
    
    if (current === 1) {
      await this.redis.expire(key, Math.ceil(windowMs / 1000));
    }
    
    return current <= limit;
  }
}

// API での使用
const rateLimiter = new RateLimiter();

export async function POST(req: Request) {
  const ip = req.headers.get('x-forwarded-for') || 'anonymous';
  const key = `rate_limit:${ip}`;
  
  const allowed = await rateLimiter.checkLimit(key, 10, 60000); // 10 requests per minute
  
  if (!allowed) {
    return NextResponse.json(
      { error: 'Rate limit exceeded' },
      { status: 429 }
    );
  }
  
  // 処理続行
}
```
